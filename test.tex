\begin{algorithm}
\KwIn{a constraint table $C_1$}
\KwIn{a variable domain $v_1$}
\KwOut{$boolean true$ if the constraint was revised, $false$ otherwise}
$revised \gets false$\;
$index \gets C_1.indexOf(v_1)$
\ForEach{$tuple \in C_1$} {
  \If{$tuple.isAlive$} {
    \If{$tuple[index] \not\in v_1$} {
      $tuple.isAlive \gets false$\;
      $C_1.numAlive \gets C_1.numAlive - 1$\;
      $revised \gets true$\;
    }
  }
}
\If{$C_1.numAlive = 0} {
  $\textbf{throw} new Error$\;
}
\Return{$revised$}\;
\caption{\textsc{ReviseConstraint}}
\end{algorithm}
